#!/bin/bash

ip=$(ifconfig | grep -Eo 'inet (addr:)?([0-9]*\.){3}[0-9]*' | grep -Eo '([0-9]*\.){3}[0-9]*' | grep -v '127.0.0.1' | cut -c 11-)

echo "IP:$ip"

for num in 1 2 3 4 5 6 7 8

do

if [ $ip -ne $num ]

then

gnome-terminal -x ./b $num

sleep 2

fi
